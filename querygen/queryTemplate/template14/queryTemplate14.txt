query multipleFilters($producerID:ID,$date:Date, $commentOfVendorKeyword:String)
{
  producer(nr:$producerID) {
    products {
      offers(where:{
        AND:[
        vendor:{publishDate:{criterion:AFTER date:$date}},
        vendor:{comment:{criterion:CONTAINS, pattern:$commentOfVendorKeyword}}
        ]}){	
          price
          offerWebpage
          product {label comment}
        }
     }
  }
}
