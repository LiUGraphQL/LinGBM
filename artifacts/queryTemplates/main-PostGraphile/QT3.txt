query researchGroup_department_head_doctorDegreeFrom($researchGroupID:ID!) 
{
  researchGroupByNr(nr: $researchGroupID) {
      departmentBySuborganizationof {
        professorsByHeadofList {
            facultyByNr {
              nr
              emailaddress
              universityByDoctoraldegreefrom {
                nr
              }
            }
        }
      }
  }
}
