query multipleFilters($universityID:ID, $age:Int, $interestkeyword:String) 
{ 
	university(nr:$universityID) { 
		undergraduateDegreeObtainedBystudent(where: { AND:{age:{pattern:$age,criterion: MORETHAN}, advisor:{researchInterest:{criterion:CONTAINS, pattern:$interestkeyword}}}}) { 
			id 
			emailAddress 
			takeGraduateCourses {id} 
		} 
	} 
}
